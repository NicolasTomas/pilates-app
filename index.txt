        async function cancelClass(classId) {
            if (!confirm("¿Deseás cancelar esta clase?")) return;

            const userRef = doc(db, "users", auth.currentUser.uid);
            const userSnap = await getDoc(userRef);
            if (!userSnap.exists()) return showAlert("Usuario no encontrado", "error");

            const userData = userSnap.data();

            // 🔹 Configuración del gimnasio
            const gymSnap = await getDoc(doc(db, "gyms", userData.gymId));
            const gymData = gymSnap.exists() ? gymSnap.data() : {};
            const minCancelHours = gymData.minCancelHours || 24;

            // 🔹 Validar tiempo mínimo (simplificado)
            const now = new Date();
            const classDoc = await getDoc(doc(db, "classes", classId));
            if (!classDoc.exists()) return showAlert("Clase no encontrada", "error");

            const classData = classDoc.data();

            // Suponemos que la clase es hoy o próxima (no hay calendario real)
            const classTime = new Date();
            const [hh, mm] = (classData.time || "00:00").split(":");
            classTime.setHours(hh, mm);

            const diffHours = (classTime - now) / 3600000;
            if (diffHours < minCancelHours) {
                return showAlert(`No podés cancelar con menos de ${minCancelHours} hs de anticipación`, "error");
            }

            // 🔹 Generar ticket
            const ticketId = "TCK_" + Date.now();
            const exp = new Date();
            exp.setDate(exp.getDate() + (gymData.creditExpiration || 30));

            const newTicket = {
                id: ticketId,
                fechaCreacion: new Date().toISOString(),
                venceEl: exp.toISOString(),
                usado: false
            };

            const tickets = userData.tickets || [];
            const maxCredits = gymData.maxCredits || 12;
            if (tickets.length >= maxCredits) {
                return showAlert("Alcanzaste el máximo de tickets disponibles", "error");
            }

            tickets.push(newTicket);

            // 🔹 Actualizar usuario y liberar camilla
            await updateDoc(userRef, {
                tickets: tickets,
                classSchedule: userData.classSchedule.filter(id => id !== classId)
            });

            // 🔹 Liberar camilla (si existe campo freeBeds)
            const free = (classData.freeBeds || 0) + 1;
            await updateDoc(doc(db, "classes", classId), { freeBeds: free });

            showAlert("✅ Clase cancelada y ticket generado", "success");
            await loadStudentPanel(userData);
        }

        async function loadRecoverClasses(userData, settings) {
            const container = document.getElementById("recoverClassesList");
            container.innerHTML = "Cargando clases...";

            const quotaExp = new Date(userData.quotaExpiration);
            const limitDate = new Date(quotaExp);
            limitDate.setDate(limitDate.getDate() + settings.autoUnsubscribe);

            // Verificar tickets
            const validTickets = (userData.tickets || []).filter(t => !t.usado && new Date(t.venceEl) > new Date());
            if (validTickets.length === 0) {
                container.innerHTML = "<p>No tenés tickets disponibles.</p>";
                return;
            }

            const qClasses = query(collection(db, "classes"), where("gymId", "==", userData.gymId));
            const snap = await getDocs(qClasses);
            container.innerHTML = "";

            for (const docSnap of snap.docs) {

                div.className = "item-card";
                div.innerHTML = `
            <h4>${(c.days || []).join(", ")} - ${c.time}</h4>
            <p>${free} camillas libres</p>
            <div class="item-actions">
                <button class="btn btn-success btn-small" onclick="recoverClass('${id}')">Recuperar</button>
            </div>
        `;
                container.appendChild(div);
            }
        }
        async function recoverClass(classId) {
            if (!confirm("¿Deseás usar un ticket para esta clase?")) return;

            const userRef = doc(db, "users", auth.currentUser.uid);
            const userSnap = await getDoc(userRef);
            const userData = userSnap.data();

            const validTickets = (userData.tickets || []).filter(t => !t.usado && new Date(t.venceEl) > new Date());
            if (validTickets.length === 0) return showAlert("No tenés tickets válidos", "error");

            const ticket = validTickets[0];
            ticket.usado = true;
            ticket.fechaUso = new Date().toISOString();

            const newTickets = [...userData.tickets.filter(t => t.id !== ticket.id), ticket];
            const newSchedule = [...(userData.classSchedule || []), classId];

            await updateDoc(userRef, {
                tickets: newTickets,
                classSchedule: newSchedule
            });

            const classRef = doc(db, "classes", classId);
            const cDoc = await getDoc(classRef);
            if (cDoc.exists()) {
                const c = cDoc.data();
                await updateDoc(classRef, { freeBeds: Math.max((c.freeBeds || 0) - 1, 0) });
            }

            showAlert("✅ Clase recuperada exitosamente", "success");
            await loadStudentPanel(userData);
        }

        window.createClass = async function () {
            const time = document.getElementById('classTime').value;
            const roomId = document.getElementById('classRoom').value || null;
            const teacherId = document.getElementById('classTeacher').value || null;
            const duration = document.getElementById('classDuration').value
                ? parseInt(document.getElementById('classDuration').value)
                : null;

            const selectedDays = Array.from(document.querySelectorAll('.day-checkbox:checked')).map(cb => cb.value);

            if (isCreatingClass) return;     // 🔒 Anti-doble clic
            isCreatingClass = true;

            if (selectedDays.length === 0) {
                alert('Seleccioná al menos un día de la semana');
                isCreatingClass = false;
                return;
            }
            if (!time) {
                alert('Por favor seleccioná un horario');
                isCreatingClass = false;
                return;
            }

            loading.style.display = 'block';

            try {
                // 🔹 Normalizamos el nombre (ordenamos días para evitar variaciones)
                const daysSorted = [...selectedDays].sort();
                const className = `Clase ${daysSorted.join(', ')} ${time}`;

                // 🔎 Chequeo client-side de duplicado (misma combinación de gym+nombre)
                // Para evitar índices compuestos, filtramos en cliente
                const qs = await getDocs(query(collection(db, "classes"), where("gymId", "==", currentUserData.gymId)));
                const already = qs.docs.some(d => {
                    const c = d.data();
                    return (c.name === className) && (c.roomId || null) === (roomId || null);
                });
                if (already) {
                    alert("⚠️ Ya existe una clase con los mismos días, horario y salón.");
                    return;
                }

                const classId = generateToken();
                await setDoc(doc(db, 'classes', classId), {
                    name: className,
                    days: daysSorted,
                    time,
                    roomId,
                    teacherId,
                    duration,
                    gymId: currentUserData.gymId,
                    createdAt: new Date().toISOString()
                });

                showAlert('✅ Clase creada exitosamente', 'success');

                // Limpiar formulario
                document.getElementById('classTime').value = '';
                document.getElementById('classDuration').value = '';
                document.querySelectorAll('.day-checkbox').forEach(cb => cb.checked = false);

                await loadClasses();
                await loadAvailableClassesForNewUser(); // para que el selector de alumno muestre el cupo

            } catch (error) {
                console.error('Error al crear clase:', error);
                alert('Error al crear clase');
            } finally {
                loading.style.display = 'none';
                isCreatingClass = false;
            }
        };

        async function loadAvailableClassesForNewUser() {
            const container = document.getElementById("availableClassesForNewUser");
            if (!container) return;
            container.innerHTML = "<p style='color:#666;'>Cargando clases...</p>";

            try {
                // Traer todas las clases del gimnasio
                const qClasses = query(collection(db, "classes"), where("gymId", "==", currentUserData.gymId));
                const snapClasses = await getDocs(qClasses);
                if (snapClasses.empty) {
                    container.innerHTML = "<p style='color:#666;'>No hay clases creadas aún.</p>";
                    return;
                }

                // Traer cantidad de alumnos por clase
                const qStudents = query(
                    collection(db, "users"),
                    where("gymId", "==", currentUserData.gymId),
                    where("role", "==", "alumno")
                );
                const snapStudents = await getDocs(qStudents);
                const ocupadas = {};

                snapStudents.forEach(stDoc => {
                    const st = stDoc.data();
                    (st.classSchedule || []).forEach(cid => {
                        ocupadas[cid] = (ocupadas[cid] || 0) + 1;
                    });
                });

                container.innerHTML = "";

                for (const docSnap of snapClasses.docs) {
                    const c = docSnap.data();
                    const id = docSnap.id;

                    // Info de salón
                    let roomName = "Sin salón";
                    let capacidad = 0;
                    if (c.roomId) {
                        const roomDoc = await getDoc(doc(db, "rooms", c.roomId));
                        if (roomDoc.exists()) {
                            const data = roomDoc.data();
                            roomName = data.name;
                            capacidad = data.capacity || 0;
                        }
                    }

                    // Info de profesor
                    let teacherName = "Sin profesor";
                    if (c.teacherId) {
                        const tDoc = await getDoc(doc(db, "users", c.teacherId));
                        if (tDoc.exists()) teacherName = tDoc.data().name;
                    }

                    // Cálculo de camillas libres
                    const ocupadasClase = ocupadas[id] || 0;
                    const libres = Math.max(0, capacidad - ocupadasClase);

                    // Si no hay lugar, deshabilitar checkbox
                    const sinLugar = libres <= 0;

                    // Crear recuadro
                    const div = document.createElement("div");
                    div.className = "class-schedule";
                    div.style.border = sinLugar ? "2px solid #f5c6cb" : "2px solid #dee2e6";
                    div.style.opacity = sinLugar ? "0.6" : "1";
                    div.style.cursor = sinLugar ? "not-allowed" : "pointer";

                    div.innerHTML = `
        <label style="display:flex;align-items:center;gap:8px;">
          <input type="checkbox" value="${id}" ${sinLugar ? "disabled" : ""} style="transform:scale(1.2);">
          <div>
            <strong>${(c.days || []).join(", ")} - ${c.time || ""}</strong><br>
            <small>🏠 ${roomName} • 👩‍🏫 ${teacherName}</small><br>
            <small>${libres} camillas libres</small>
          </div>
        </label>
      `;

                    container.appendChild(div);
                }

            } catch (error) {
                console.error("Error al cargar clases para nuevo usuario:", error);
                container.innerHTML = "<p style='color:red;'>Error al cargar las clases.</p>";
            }
        }


        // ======== CREAR USUARIO (Alumnos y Profesores) con cupo y anti-doble click ======== //
        let isCreatingUser = false;

        window.createUserDirect = async function () {
            const msgBox = document.getElementById("createUserMessage");
            msgBox.textContent = "";
            msgBox.style.display = "none";

            if (isCreatingUser) return;          // 🔒 Anti-doble clic
            isCreatingUser = true;

            const role = document.getElementById("newUserRole").value;
            const firstName = document.getElementById("newUserFirstName").value.trim();
            const lastName = document.getElementById("newUserLastName").value.trim();
            const dni = document.getElementById("newUserDNI").value.trim();
            const phone = document.getElementById("newUserPhone").value.trim();
            const emergency = document.getElementById("newUserEmergency").value.trim();

            const selectedClassCheckboxes = document.querySelectorAll("#availableClassesForNewUser input[type='checkbox']:checked");
            const selectedClassIds = Array.from(selectedClassCheckboxes).map(cb => cb.value);

            const showError = (txt) => {
                msgBox.innerHTML = txt;
                msgBox.style.color = "#c82333";
                msgBox.style.display = "block";
            };
            const showOk = (txt) => {
                msgBox.innerHTML = txt;
                msgBox.style.color = "#155724";
                msgBox.style.display = "block";
            };

            if (!firstName || !lastName || !dni || !phone || !emergency) {
                showError("⚠️ Completá todos los campos obligatorios.");
                isCreatingUser = false;
                return;
            }
            if (!/^[0-9]+$/.test(dni)) {
                showError("⚠️ El DNI debe contener solo números.");
                isCreatingUser = false;
                return;
            }
            if (role === "alumno" && selectedClassIds.length === 0) {
                showError("⚠️ Seleccioná al menos una clase para el alumno.");
                isCreatingUser = false;
                return;
            }

            loading.style.display = "block";

            try {
                // 1) Evitar duplicado por DNI
                const existsSnap = await getDocs(
                    query(collection(db, "users"), where("dni", "==", dni))
                );
                if (!existsSnap.empty) {
                    showError(`⚠️ Ya existe un usuario con el DNI ${dni}.`);
                    return;
                }

                // 2) Si es alumno, validar CUPOS antes de crear
                if (role === "alumno") {
                    // Ocupación por clase
                    const occupancy = await buildClassOccupancyMap();
                    // Necesitamos la capacidad del salón de cada clase
                    for (const classId of selectedClassIds) {
                        const classDoc = await getDoc(doc(db, "classes", classId));
                        if (!classDoc.exists()) {
                            showError("⚠️ Una de las clases seleccionadas ya no existe.");
                            return;
                        }
                        const c = classDoc.data();
                        let capacity = 0;
                        if (c.roomId) {
                            const roomDoc = await getDoc(doc(db, "rooms", c.roomId));
                            if (roomDoc.exists()) capacity = roomDoc.data().capacity || 0;
                        }
                        const occupied = occupancy[classId] || 0;
                        const free = Math.max(capacity - occupied, 0);
                        if (free <= 0) {
                            showError(`⚠️ La clase "${(c.days || []).join(", ")} ${c.time || ""}" no tiene cupos.`);
                            return;
                        }
                    }
                }

                // 3) Crear usuario en Auth con app secundaria (sin patearte la sesión)
                const fakeEmail = `${dni}@fakeuser.com`;
                const secondaryApp = initializeApp(firebaseConfig, "secondaryApp_" + Date.now());
                const secondaryAuth = getAuth(secondaryApp);
                const password = dni;
                const userCredential = await createUserWithEmailAndPassword(secondaryAuth, fakeEmail, password);
                const uid = userCredential.user.uid;

                // 4) Grabar datos en Firestore
                const userData = {
                    name: `${firstName} ${lastName}`,
                    firstName, lastName,
                    dni, phone, emergency,
                    email: fakeEmail,
                    role,
                    gymId: currentUserData.gymId,
                    createdAt: new Date().toISOString(),
                };

                if (role === "alumno") {
                    const nextMonth = new Date();
                    nextMonth.setMonth(nextMonth.getMonth() + 1);
                    userData.quotaExpiration = nextMonth.toISOString();
                    userData.credits = 4;
                    userData.classSchedule = selectedClassIds;
                }

                await setDoc(doc(db, "users", uid), userData);
                await signOut(secondaryAuth);

                // 5) UI
                showOk(`✅ ${role === "alumno" ? "Alumno" : "Profesor"} creado correctamente.`);
                document.getElementById("newUserFirstName").value = "";
                document.getElementById("newUserLastName").value = "";
                document.getElementById("newUserDNI").value = "";
                document.getElementById("newUserPhone").value = "";
                document.getElementById("newUserEmergency").value = "";
                document.querySelectorAll("#availableClassesForNewUser input[type='checkbox']").forEach(cb => cb.checked = false);

                // 6) Refrescar tablas y camillas libres
                await loadStudents();
                await loadClasses();

            } catch (error) {
                console.error("❌ Error creando usuario:", error);
                if (error.code === "auth/email-already-in-use") {
                    showError(`⚠️ Ya existe un usuario con el DNI ${dni}.`);
                } else {
                    showError("❌ No se pudo crear el usuario. Intentalo nuevamente.");
                }
            } finally {
                loading.style.display = "none";
                isCreatingUser = false;
            }
        };

            const description = document.getElementById('settingsDescription').value.trim();

            if (!name) {
                alert('El nombre del gimnasio es obligatorio');
                return;
            }

            loading.style.display = 'block';

            try {
                await updateDoc(doc(db, 'gyms', currentUserData.gymId), {
                    name: name,
                    address: address,
                    phone: phone,
                    email: email,
                    description: description,
                    updatedAt: new Date().toISOString()
                });

                document.getElementById('displayGymName').textContent = name;

            } catch (error) {
                console.error('Error:', error);
                alert('Error al guardar información');
            } finally {
                loading.style.display = 'none';
            }
        };

        window.changePassword = async function () {
            const currentPassword = document.getElementById('currentPassword').value;
            const newPassword = document.getElementById('newPassword').value;
            const confirmPassword = document.getElementById('confirmPassword').value;

            if (!currentPassword || !newPassword || !confirmPassword) {
                alert('Por favor completa todos los campos');
                return;
            }

            if (newPassword !== confirmPassword) {
                alert('Las contraseñas nuevas no coinciden');
                return;
            }

            if (newPassword.length < 6) {
                alert('La contraseña debe tener al menos 6 caracteres');
                return;
            }

            loading.style.display = 'block';

            try {
                const credential = EmailAuthProvider.credential(
                    auth.currentUser.email,
                    currentPassword
                );
                await reauthenticateWithCredential(auth.currentUser, credential);
                await updatePassword(auth.currentUser, newPassword);

                alert('✅ Contraseña actualizada exitosamente');
                document.getElementById('currentPassword').value = '';
                document.getElementById('newPassword').value = '';
                document.getElementById('confirmPassword').value = '';

            } catch (error) {
                console.error('Error:', error);
                if (error.code === 'auth/wrong-password') {
                    alert('La contraseña actual es incorrecta');
                } else {
                    alert('Error al cambiar contraseña: ' + error.message);
                }
            } finally {
                loading.style.display = 'none';
            }
        };

        onAuthStateChanged(auth, async (user) => {
            try {
                const inviteToken = new URLSearchParams(window.location.search).get('invite');

                // Si hay un link de invitación, no interferir
                if (inviteToken) {
                    console.log("🔒 Modo invitación activo. Ignorando sesión previa.");
                    await signOut(auth);
                    return;
                }

                if (user) {
                    const userDoc = await getDoc(doc(db, 'users', user.uid));
                    if (userDoc.exists()) {
                        currentUserData = userDoc.data();

                        // 🔹 Mostrar el dashboard según el rol
                        dashboard.dataset.loaded = "false";
                        await showDashboard(currentUserData);
                    } else {
                        console.warn("⚠️ Usuario sin documento en Firestore.");
                        await signOut(auth);
                        loginForm.style.display = 'block';
                    }
                } else {
                    // 🔹 Si no hay usuario autenticado, mostrar login
                    dashboard.classList.remove('active');
                    loginForm.style.display = 'block';
                }
            } catch (error) {
                console.error("❌ Error en onAuthStateChanged:", error);
            }
        });






        // 🔁 Recalcular camillas libres por clase
        async function recalculateClassFreeBeds(classId) {
            try {
                const occupancy = await buildClassOccupancyMap();
                const classDoc = await getDoc(doc(db, "classes", classId));
                if (!classDoc.exists()) return;

                const c = classDoc.data();
                let capacity = 0;
                if (c.roomId) {
                    const roomDoc = await getDoc(doc(db, "rooms", c.roomId));
                    if (roomDoc.exists()) capacity = roomDoc.data().capacity || 0;
                }

                const occupied = occupancy[classId] || 0;
                const free = Math.max(capacity - occupied, 0);

                await updateDoc(doc(db, "classes", classId), { freeBeds: free });
                console.log(`✅ Clase ${classId}: ${free} camillas libres (${capacity} totales, ${occupied} ocupadas)`);
            } catch (error) {
                console.error("Error al recalcular camillas libres:", error);
            }
        }
        // ======== CARGA AUTOMÁTICA DE SALONES Y PROFESORES AL CREAR CLASE ======== //
        async function loadClassCreationOptions() {
            try {
                // --- Salones ---
                const roomSelect = document.getElementById("classRoom");
                if (roomSelect) {
                    roomSelect.innerHTML = "<option value=''>Seleccionar salón</option>";
                    const roomSnap = await getDocs(
                        query(collection(db, "rooms"), where("gymId", "==", currentUserData.gymId))
                    );

                    if (roomSnap.empty) {
                        const opt = document.createElement("option");
                        opt.value = "";
                        opt.textContent = "⚠️ No hay salones creados";
                        roomSelect.appendChild(opt);
                    } else {
                        roomSnap.forEach(r => {
                            const data = r.data();
                            const opt = document.createElement("option");
                            opt.value = r.id;
                            opt.textContent = data.name;
                            roomSelect.appendChild(opt);
                        });
                    }
                }

                // --- Profesores ---
                const teacherSelect = document.getElementById("classTeacher");
                if (teacherSelect) {
                    teacherSelect.innerHTML = "<option value=''>Seleccionar profesor</option>";
                    const teacherSnap = await getDocs(
                        query(
                            collection(db, "users"),
                            where("gymId", "==", currentUserData.gymId),
                            where("role", "==", "profesor")
                        )
                    );

                    if (teacherSnap.empty) {
                        const opt = document.createElement("option");
                        opt.value = "";
                        opt.textContent = "⚠️ No hay profesores registrados";
                        teacherSelect.appendChild(opt);
                    } else {
                        teacherSnap.forEach(t => {
                            const data = t.data();
                            const opt = document.createElement("option");
                            opt.value = t.id;
                            opt.textContent = data.name;
                            teacherSelect.appendChild(opt);
                        });
                    }
                }
            } catch (error) {
                console.error("Error al cargar salones/profesores:", error);
            }
        }
